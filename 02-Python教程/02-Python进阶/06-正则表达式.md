# 正则表达式

## 6.1 正则表达式基础

正则表达式（Regular Expression）是一种用于匹配字符串中字符组合的模式。在安全工具中，常用于日志分析、数据提取、参数校验等场景。

### re 模块核心函数

```python
import re

text = "Hello, my email is alice@example.com and phone is 138-1234-5678"

# match —— 从字符串开头匹配
m = re.match(r"Hello", text)
print(m.group() if m else "无匹配")  # Hello

# search —— 搜索整个字符串，返回第一个匹配
m = re.search(r"\d{3}-\d{4}-\d{4}", text)
print(m.group())  # 138-1234-5678

# findall —— 查找所有匹配，返回列表
emails = re.findall(r"\w+@\w+\.\w+", text)
print(emails)  # ['alice@example.com']

# finditer —— 查找所有匹配，返回迭代器
for m in re.finditer(r"\w+@\w+\.\w+", text):
    print(f"找到邮箱：{m.group()}，位置：{m.span()}")

# sub —— 替换匹配的文本
result = re.sub(r"\d{3}-\d{4}-\d{4}", "[PHONE]", text)
print(result)  # Hello, my email is alice@example.com and phone is [PHONE]

# split —— 按匹配分割字符串
parts = re.split(r"\s+", text)
print(parts)  # 按空白字符分割
```

## 6.2 字符与量词

### 元字符

| 元字符 | 含义 |
|--------|------|
| `.` | 匹配任意单个字符（除换行） |
| `\d` | 匹配数字 `[0-9]` |
| `\D` | 匹配非数字 |
| `\w` | 匹配字母、数字、下划线 `[a-zA-Z0-9_]` |
| `\W` | 匹配非单词字符 |
| `\s` | 匹配空白字符（空格、\t、\n等） |
| `\S` | 匹配非空白字符 |

### 字符集

```python
text = "Hello 123 World"

# [abc] —— 匹配a或b或c
re.findall(r"[aeiou]", text)  # ['e', 'o']

# [a-z] —— 匹配小写字母
re.findall(r"[a-z]", text)    # ['e', 'l', 'l', 'o', 'o', 'r', 'l', 'd']

# [0-9] —— 匹配数字
re.findall(r"[0-9]", text)    # ['1', '2', '3']

# [^abc] —— 匹配除a、b、c之外的字符
re.findall(r"[^a-zA-Z\s]", text)  # ['1', '2', '3']
```

### 量词

| 量词 | 含义 |
|------|------|
| `*` | 前一个字符出现0次或多次 |
| `+` | 前一个字符出现1次或多次 |
| `?` | 前一个字符出现0次或1次 |
| `{n}` | 前一个字符恰好出现n次 |
| `{n,}` | 前一个字符至少出现n次 |
| `{n,m}` | 前一个字符出现n到m次 |

```python
text = "abc, ac, abbc, abbbc"

# * —— 0次或多次
re.findall(r"ab*c", text)   # ['abc', 'ac', 'abbc', 'abbbc']

# + —— 1次或多次
re.findall(r"ab+c", text)   # ['abc', 'abbc', 'abbbc']

# ? —— 0次或1次
re.findall(r"ab?c", text)   # ['abc', 'ac']

# 精确控制
re.findall(r"ab{2}c", text)   # ['abbc']（b恰好2次）
re.findall(r"ab{2,}c", text)  # ['abbc', 'abbbc']（b至少2次）
re.findall(r"ab{1,3}c", text) # ['abc', 'abbc', 'abbbc']（b出现1-3次）
```

## 6.3 分组与捕获

使用 `()` 将模式分组，可以提取子匹配：

```python
text = "2026-06-23 15:30:00"
pattern = r"(\d{4})-(\d{2})-(\d{2}) (\d{2}):(\d{2}):(\d{2})"
match = re.search(pattern, text)

if match:
    print("完整匹配：", match.group(0))   # 2026-06-23 15:30:00
    print("年份：", match.group(1))        # 2026
    print("月份：", match.group(2))        # 06
    print("日期：", match.group(3))        # 23
    print("所有分组：", match.groups())    # ('2026', '06', '23', '15', '30', '00')
```

### 命名分组

```python
pattern = r"(?P<year>\d{4})-(?P<month>\d{2})-(?P<day>\d{2})"
match = re.search(pattern, "2026-06-23")
if match:
    print(match.group("year"))   # 2026
    print(match.group("month"))  # 06
    print(match.group("day"))    # 23
    print(match.groupdict())     # {'year': '2026', 'month': '06', 'day': '23'}
```

## 6.4 边界匹配

| 边界符号 | 含义 |
|---------|------|
| `^` | 字符串开头 |
| `$` | 字符串结尾 |
| `\b` | 单词边界 |
| `\B` | 非单词边界 |

```python
# ^ 和 $ —— 整行匹配
re.match(r"^\d+$", "12345")     # 匹配（整行都是数字）
re.match(r"^\d+$", "123abc")    # 不匹配

# \b —— 单词边界
text = "cat category scatter cat"
re.findall(r"\bcat\b", text)    # ['cat', 'cat']（只匹配完整的单词"cat"）
re.findall(r"cat", text)        # ['cat', 'cat', 'cat', 'cat']（所有包含cat的）
```

## 6.5 贪婪与非贪婪

默认情况下，量词是贪婪的（尽量匹配更多字符）。在量词后加 `?` 可变为非贪婪：

```python
text = "<div>内容1</div><div>内容2</div>"

# 贪婪匹配
greedy = re.findall(r"<div>.*</div>", text)
print(greedy)  # ['<div>内容1</div><div>内容2</div>']

# 非贪婪匹配
non_greedy = re.findall(r"<div>.*?</div>", text)
print(non_greedy)  # ['<div>内容1</div>', '<div>内容2</div>']
```

## 6.6 常见安全场景中的正则

### IP地址匹配

```python
ip_pattern = r"^(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}(?:25[0-5]|2[0-4]\d|[01]?\d\d?)$"

def is_valid_ip(ip):
    return bool(re.match(ip_pattern, ip))

print(is_valid_ip("192.168.1.1"))    # True
print(is_valid_ip("256.1.2.3"))      # False
print(is_valid_ip("10.0.0.1"))       # True
```

### URL提取

```python
url_pattern = r"https?://[^\s<>\"']+|www\.[^\s<>\"']+"

text = "我的网站是 https://example.com/path?q=1，备用是 http://test.org"
urls = re.findall(url_pattern, text)
print(urls)  # ['https://example.com/path?q=1', 'http://test.org']
```

### HTTP日志解析

```python
log_line = '192.168.1.1 - - [23/Jun/2026:15:30:00 +0800] "GET /admin/login.php HTTP/1.1" 200 5234 "https://example.com" "Mozilla/5.0"'

# 解析Apache/Nginx日志模式
log_pattern = r'^(\S+) \S+ \S+ \[([^\]]+)\] "(\w+) (\S+) [^"]*" (\d{3}) (\d+)'
match = re.search(log_pattern, log_line)

if match:
    print(f"IP: {match.group(1)}")
    print(f"时间: {match.group(2)}")
    print(f"请求方法: {match.group(3)}")
    print(f"请求路径: {match.group(4)}")
    print(f"状态码: {match.group(5)}")
    print(f"响应大小: {match.group(6)} 字节")
```

### SQL注入检测

```python
def detect_sql_injection(input_text):
    """
    检测输入中是否包含SQL注入特征
    """
    sql_patterns = [
        r"(?i)\bunion\s+select\b",
        r"(?i)\bselect\s+.*\bfrom\b",
        r"(?i)or\s+1\s*=\s*1",
        r"(?i)'.*or.*'",
        r"--\s*$",
        r"/\*.*\*/",
        r"(?i)exec\s*\(",
        r"(?i)char\s*\(",
        r"(?i)waitfor\s+delay",
    ]
    
    for pattern in sql_patterns:
        if re.search(pattern, input_text):
            return True, f"检测到SQL注入特征：{pattern}"
    return False, "未检测到注入特征"

# 测试
tests = [
    "1' OR '1'='1",
    "SELECT * FROM users",
    "hello world",
    "admin' --",
    "a = 1 and 1 = 1",
]
for test in tests:
    detected, msg = detect_sql_injection(test)
    print(f"[{'!' if detected else ' '}] {test:20} -> {msg}")
```

## 6.7 编译正则

多次使用同一正则时，预编译可以提高性能：

```python
import re
import time

# 不编译（每次重新解析）
start = time.time()
for _ in range(10000):
    re.findall(r"\d{3}-\d{4}-\d{4}", "138-1234-5678")
print(f"未编译：{time.time() - start:.3f}s")

# 预编译
pattern = re.compile(r"\d{3}-\d{4}-\d{4}")
start = time.time()
for _ in range(10000):
    pattern.findall("138-1234-5678")
print(f"已编译：{time.time() - start:.3f}s")
```

## 6.8 本章练习

```python
import re

# 1. 提取HTML中的所有链接
def extract_links(html):
    pattern = r'href=["\'](https?://[^"\']+)["\']'
    return re.findall(pattern, html)

# 2. 验证密码强度
def password_strength(password):
    """检查密码强度并给出评分（满分5分）"""
    score = 0
    if len(password) >= 8: score += 1
    if re.search(r"[a-z]", password): score += 1
    if re.search(r"[A-Z]", password): score += 1
    if re.search(r"\d", password): score += 1
    if re.search(r"[!@#$%^&*(),.?\":{}|<>]", password): score += 1
    return score

# 3. 提取URL中的参数
def extract_url_params(url):
    """从URL中提取查询参数"""
    match = re.search(r"\?(.+)$", url)
    if not match:
        return {}
    params = {}
    for pair in match.group(1).split("&"):
        if "=" in pair:
            key, value = pair.split("=", 1)
            params[key] = value
    return params

# 4. 替换敏感信息脱敏
def mask_sensitive(text):
    """对文本中的敏感信息进行脱敏处理"""
    # 脱敏手机号：138****5678
    text = re.sub(r"(1[3-9]\d)\d{4}(\d{4})", r"\1****\2", text)
    # 脱敏邮箱：a***@example.com
    text = re.sub(r"(\w)\w+@", r"\1***@", text)
    return text
```
