# 网络扫描器开发

## 1.1 扫描器设计思路

网络扫描器用于探测目标网络中的活跃主机和开放服务。其核心功能包括：

- 主机发现（Ping扫描）
- 端口状态检测
- 服务指纹识别
- 操作系统识别

### 整体架构

```
Scanner
├── PingScanner    -- 主机存活检测
├── PortScanner    -- 端口扫描
├── ServiceDetector -- 服务识别
└── Report         -- 结果输出
```

## 1.2 主机存活检测

```python
import subprocess
import platform
import socket
import concurrent.futures

class PingScanner:
    """使用Ping进行主机存活检测"""
    
    def __init__(self, timeout=2):
        self.timeout = timeout
        # 根据操作系统选择ping参数
        self.param = "-n" if platform.system().lower() == "windows" else "-c"
    
    def ping(self, ip):
        """对单个IP执行ping检测"""
        try:
            cmd = ["ping", self.param, "1", "-w", str(self.timeout * 1000), ip]
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=self.timeout + 1
            )
            if result.returncode == 0:
                return ip, True
            return ip, False
        except (subprocess.TimeoutExpired, Exception):
            return ip, False
    
    def scan_network(self, ip_list):
        """并发扫描多个IP"""
        alive = []
        with concurrent.futures.ThreadPoolExecutor(max_workers=20) as executor:
            futures = {executor.submit(self.ping, ip): ip for ip in ip_list}
            for future in concurrent.futures.as_completed(futures):
                ip, is_alive = future.result()
                if is_alive:
                    alive.append(ip)
                    print(f"[存活] {ip}")
        return alive
    
    @staticmethod
    def generate_ip_range(cidr):
        """
        从CIDR生成IP地址列表（简化版）
        示例：192.168.1.0/24
        """
        from ipaddress import ip_network
        network = ip_network(cidr, strict=False)
        return [str(ip) for ip in network.hosts()]

# 使用示例
if __name__ == "__main__":
    scanner = PingScanner()
    ips = ["192.168.1.1", "192.168.1.2", "8.8.8.8", "114.114.114.114"]
    alive_hosts = scanner.scan_network(ips)
    print(f"\n存活主机：{len(alive_hosts)} 个")
    for ip in alive_hosts:
        print(f"  - {ip}")
```

## 1.3 TCP连接扫描

```python
import socket
import threading
import time

class TCPConnectScanner:
    """
    TCP全连接扫描器
    
    通过完整的三次握手判断端口状态：
    - connect() 成功 → 端口开放
    - connect() 失败（refused）→ 端口关闭
    - 超时 → 端口被过滤
    """
    
    def __init__(self, host, timeout=2, max_threads=100):
        self.host = host
        self.timeout = timeout
        self.max_threads = max_threads
        self.open_ports = []
        self.lock = threading.Lock()
    
    def scan_port(self, port):
        """扫描单个端口"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            
            # connect_ex 返回0表示连接成功
            result = sock.connect_ex((self.host, port))
            sock.close()
            
            if result == 0:
                with self.lock:
                    self.open_ports.append(port)
                return port, "open"
            return port, "closed"
        
        except socket.gaierror:
            return port, "unknown_host"
        except Exception as e:
            return port, f"error: {e}"
    
    def scan_range(self, start_port, end_port):
        """扫描端口范围（使用信号量控制并发）"""
        semaphore = threading.Semaphore(self.max_threads)
        threads = []
        
        def worker(port):
            semaphore.acquire()
            self.scan_port(port)
            semaphore.release()
        
        start_time = time.time()
        
        for port in range(start_port, end_port + 1):
            t = threading.Thread(target=worker, args=(port,))
            threads.append(t)
            t.start()
        
        for t in threads:
            t.join()
        
        duration = time.time() - start_time
        return sorted(self.open_ports), duration

# 使用示例
if __name__ == "__main__":
    scanner = TCPConnectScanner("scanme.nmap.org", timeout=3, max_threads=200)
    open_ports, duration = scanner.scan_range(1, 1024)
    
    print(f"扫描耗时：{duration:.2f}秒")
    print(f"开放端口（共{len(open_ports)}个）：")
    for port in open_ports:
        print(f"  {port}/tcp")
```

## 1.5 服务指纹识别

```python
import socket

class ServiceDetector:
    """通过横幅抓取识别服务"""
    
    SERVICE_PORTS = {
        21: "FTP", 22: "SSH", 23: "Telnet",
        25: "SMTP", 53: "DNS", 80: "HTTP",
        110: "POP3", 143: "IMAP", 443: "HTTPS",
        445: "SMB", 1433: "MSSQL", 1521: "Oracle",
        3306: "MySQL", 3389: "RDP", 5432: "PostgreSQL",
        6379: "Redis", 8080: "HTTP-Proxy", 27017: "MongoDB"
    }
    
    def __init__(self, host, timeout=3):
        self.host = host
        self.timeout = timeout
    
    def grab_banner(self, port):
        """尝试抓取服务横幅"""
        service = self.SERVICE_PORTS.get(port, "unknown")
        banner = None
        
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            sock.connect((self.host, port))
            
            # 某些服务需要发送数据才会响应
            if port in [80, 8080]:
                sock.send(b"GET / HTTP/1.0\r\n\r\n")
            elif port in [21]:
                pass  # FTP会主动发送横幅
            
            try:
                banner = sock.recv(1024).decode("utf-8", errors="ignore").strip()
            except socket.timeout:
                pass
            
            sock.close()
        except Exception:
            pass
        
        return {
            "port": port,
            "service": service,
            "banner": banner[:100] if banner else None
        }

# 使用示例
if __name__ == "__main__":
    detector = ServiceDetector("127.0.0.1")
    results = []
    for port in [80, 22, 3306, 6379]:
        info = detector.grab_banner(port)
        if info:
            results.append(info)
            print(f"Port {port}: {info['service']} - {info['banner'] or '无横幅'}")
```

## 1.6 综合扫描器实现

```python
import socket
import threading
import time
import json

class NetworkScanner:
    """综合网络扫描器"""
    
    def __init__(self, host, timeout=2, max_threads=100):
        self.host = host
        self.timeout = timeout
        self.max_threads = max_threads
        self.results = []
        self.lock = threading.Lock()
    
    def resolve_host(self):
        """解析主机名"""
        try:
            ip = socket.gethostbyname(self.host)
            print(f"目标解析：{self.host} -> {ip}")
            return ip
        except socket.gaierror:
            print(f"无法解析主机：{self.host}")
            return None
    
    def scan_port(self, port):
        """扫描单个端口并识别服务"""
        ip = self.resolve_host()
        if not ip:
            return
        
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(self.timeout)
        
        try:
            result = sock.connect_ex((ip, port))
            if result == 0:
                info = {
                    "port": port,
                    "state": "open",
                    "service": self._guess_service(port)
                }
                
                # 尝试抓取横幅
                try:
                    if port in [80, 8080]:
                        sock.send(b"GET / HTTP/1.0\r\n\r\n")
                    banner = sock.recv(256).decode("utf-8", errors="ignore").strip()
                    info["banner"] = banner[:100]
                except:
                    pass
                
                with self.lock:
                    self.results.append(info)
                
                print(f"[开放] {port}/tcp - {info['service']}")
        
        except Exception:
            pass
        finally:
            sock.close()
    
    def _guess_service(self, port):
        """根据端口猜测服务"""
        services = {
            20: "FTP-data", 21: "FTP", 22: "SSH", 23: "Telnet",
            25: "SMTP", 53: "DNS", 80: "HTTP", 110: "POP3",
            143: "IMAP", 443: "HTTPS", 445: "SMB", 993: "IMAPS",
            995: "POP3S", 1433: "MSSQL", 1521: "Oracle", 3306: "MySQL",
            3389: "RDP", 5432: "PostgreSQL", 5900: "VNC", 6379: "Redis",
            8080: "HTTP-proxy", 8443: "HTTPS-alt", 27017: "MongoDB"
        }
        return services.get(port, "unknown")
    
    def scan(self, ports=None):
        """
        执行扫描
        ports: 端口列表，默认为常见端口
        """
        if ports is None:
            common_ports = [
                21, 22, 23, 25, 53, 80, 110, 135, 139, 143,
                443, 445, 993, 995, 1433, 1521, 3306, 3389,
                5432, 5900, 6379, 8080, 8443, 27017
            ]
            ports = common_ports
        
        print(f"开始扫描 {self.host}")
        print(f"待扫描端口数：{len(ports)}")
        start_time = time.time()
        
        threads = []
        for port in ports:
            t = threading.Thread(target=self.scan_port, args=(port,))
            threads.append(t)
            t.start()
            
            # 控制并发数
            if len([t for t in threads if t.is_alive()]) >= self.max_threads:
                time.sleep(0.01)
        
        for t in threads:
            t.join()
        
        duration = time.time() - start_time
        self.results.sort(key=lambda x: x["port"])
        
        print(f"\n=== 扫描报告 ===")
        print(f"目标：{self.host}")
        print(f"耗时：{duration:.2f}秒")
        print(f"开放端口：{len(self.results)} 个")
        
        for r in self.results:
            banner = f" - {r['banner']}" if r.get('banner') else ""
            print(f"  {r['port']}/tcp  {r['service']}{banner}")
        
        return self.results
    
    def export_json(self, filename="scan_result.json"):
        """导出为JSON格式"""
        with open(filename, "w", encoding="utf-8") as f:
            json.dump({
                "target": self.host,
                "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
                "results": self.results
            }, f, ensure_ascii=False, indent=2)
        print(f"结果已导出至 {filename}")

# 主程序
if __name__ == "__main__":
    import sys
    
    target = sys.argv[1] if len(sys.argv) > 1 else "scanme.nmap.org"
    scanner = NetworkScanner(target, timeout=2, max_threads=50)
    results = scanner.scan()
    scanner.export_json()
```
