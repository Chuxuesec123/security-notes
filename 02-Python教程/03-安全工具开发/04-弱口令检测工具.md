# 弱口令检测工具

## 4.1 弱口令检测概述

弱口令是网络安全中最常见的漏洞之一。弱口令检测工具通过尝试常见的用户名和密码组合来验证目标服务是否存在弱口令。

### 常见弱口令类型

- 默认密码（admin/admin, root/root）
- 常见弱密码（123456, password, qwerty）
- 空密码或与用户名相同
- 简单数字序列

### 支持的服务类型

| 服务 | 端口 | 协议 |
|------|------|------|
| SSH | 22 | TCP |
| FTP | 21 | TCP |
| MySQL | 3306 | TCP |
| Redis | 6379 | TCP |
| SMB | 445 | TCP |
| RDP | 3389 | TCP |

## 4.2 SSH弱口令检测

```python
import socket
import paramiko
import threading
import time
from concurrent.futures import ThreadPoolExecutor

class SSHCracker:
    """SSH弱口令检测"""
    
    # 常见用户名/密码字典
    COMMON_CREDENTIALS = [
        ("root", "root"),
        ("root", "admin"),
        ("root", "123456"),
        ("root", "password"),
        ("root", "toor"),
        ("admin", "admin"),
        ("admin", "admin123"),
        ("admin", "password"),
        ("admin", "123456"),
        ("test", "test"),
        ("user", "user"),
        ("user", "password"),
        ("guest", "guest"),
        ("www", "www"),
        ("web", "web"),
    ]
    
    def __init__(self, host, port=22, timeout=5):
        self.host = host
        self.port = port
        self.timeout = timeout
        self.found = []
        self.lock = threading.Lock()
    
    def check_credential(self, username, password):
        """尝试一组用户名密码"""
        try:
            client = paramiko.SSHClient()
            client.set_missing_host_key_policy(paramiko.AutoAddPolicy())
            client.connect(
                self.host,
                port=self.port,
                username=username,
                password=password,
                timeout=self.timeout,
                banner_timeout=self.timeout
            )
            client.close()
            
            with self.lock:
                self.found.append((username, password))
            
            return True, username, password
        
        except paramiko.AuthenticationException:
            return False, username, password
        except socket.timeout:
            return None, username, password  # 超时重试
        except Exception as e:
            return None, username, password
    
    def crack(self, wordlist=None, max_workers=10):
        """
        开始弱口令检测
        
        Args:
            wordlist: 自定义字典，格式为 [(user, pass), ...]
            max_workers: 最大并发数
        """
        credentials = wordlist or self.COMMON_CREDENTIALS
        
        print(f"SSH弱口令检测：{self.host}:{self.port}")
        print(f"字典大小：{len(credentials)}")
        print("-" * 40)
        
        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = []
            for username, password in credentials:
                future = executor.submit(
                    self.check_credential, username, password
                )
                futures.append((future, username, password))
            
            for future, username, password in futures:
                success, user, pwd = future.result()
                
                if success:
                    print(f"[成功] {user}:{pwd}")
                elif success is None:
                    pass  # 超时忽略
                else:
                    print(f"[失败] {user}:{pwd}", end="\r")
        
        print("\n" + "-" * 40)
        if self.found:
            print(f"发现 {len(self.found)} 组有效凭据：")
            for user, pwd in self.found:
                print(f"  {user}:{pwd}")
        else:
            print("未发现弱口令")
        
        return self.found


# 使用示例
def ssh_brute_force(host, port=22):
    cracker = SSHCracker(host, port)
    return cracker.crack()
```

## 4.3 FTP弱口令检测

```python
import ftplib
from concurrent.futures import ThreadPoolExecutor

class FTPCracker:
    """FTP弱口令检测"""
    
    COMMON_CREDENTIALS = [
        ("anonymous", "anonymous"),
        ("anonymous", ""),
        ("ftp", "ftp"),
        ("admin", "admin"),
        ("admin", "123456"),
        ("admin", "password"),
        ("root", "root"),
        ("test", "test"),
        ("user", "user"),
        ("user", "123456"),
    ]
    
    def __init__(self, host, port=21, timeout=10):
        self.host = host
        self.port = port
        self.timeout = timeout
        self.found = []
    
    def check_credential(self, username, password):
        """尝试FTP登录"""
        try:
            ftp = ftplib.FTP()
            ftp.connect(self.host, self.port, self.timeout)
            ftp.login(username, password)
            ftp.quit()
            return True, username, password
        except ftplib.error_perm:
            return False, username, password
        except Exception:
            return None, username, password
    
    def crack(self, wordlist=None, max_workers=5):
        credentials = wordlist or self.COMMON_CREDENTIALS
        
        print(f"FTP弱口令检测：{self.host}:{self.port}")
        
        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = {
                executor.submit(self.check_credential, u, p): (u, p)
                for u, p in credentials
            }
            
            for future in futures:
                success, username, password = future.result()
                if success:
                    self.found.append((username, password))
                    print(f"[成功] {username}:{password}")
                elif success is None:
                    print(f"[超时] {username}:{password}")
                else:
                    print(f"[失败] {username}:{password}", end="\r")
        
        return self.found
```

## 4.5 Redis弱口令检测

```python
import socket

class RedisCracker:
    """Redis弱口令检测"""
    
    COMMON_PASSWORDS = [
        "", "redis", "root", "admin",
        "123456", "password", "test"
    ]
    
    def __init__(self, host, port=6379, timeout=5):
        self.host = host
        self.port = port
        self.timeout = timeout
    
    def check_credential(self, password):
        """尝试Redis密码"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            sock.connect((self.host, self.port))
            
            if password:
                cmd = f"AUTH {password}\r\n"
            else:
                cmd = "PING\r\n"  # 无密码时直接PING
            
            sock.send(cmd.encode())
            response = sock.recv(1024).decode()
            sock.close()
            
            # +OK 表示成功
            if response.startswith("+OK") or "+OK" in response:
                return True, password
            return False, password
        
        except Exception as e:
            return None, password
    
    def crack(self):
        print(f"Redis弱口令检测：{self.host}:{self.port}")
        
        for password in self.COMMON_PASSWORDS:
            success, pwd = self.check_credential(password)
            if success:
                if pwd:
                    print(f"[成功] 密码：{pwd}")
                else:
                    print("[成功] 未设置密码！")
                return True, password
            elif success is None:
                print(f"[超时] {self.host}:{self.port}")
                break
            else:
                pwd_display = pwd if pwd else "(空)"
                print(f"[失败] 密码：{pwd_display}", end="\r")
        
        print("\n未发现弱口令")
        return False, None
```

## 4.6 综合弱口令检测工具

```python
import json
import time
import sys

class WeakPasswordDetector:
    """综合弱口令检测器"""
    
    SUPPORTED_SERVICES = {
        "ssh": {
            "port": 22,
            "class": SSHCracker,
            "description": "SSH远程登录"
        },
        "ftp": {
            "port": 21,
            "class": FTPCracker,
            "description": "FTP文件传输"
        },
        "mysql": {
            "port": 3306,
            "class": MySQLCracker,
            "description": "MySQL数据库"
        },
        "redis": {
            "port": 6379,
            "class": RedisCracker,
            "description": "Redis缓存"
        }
    }
    
    def __init__(self, host, timeout=5):
        self.host = host
        self.timeout = timeout
        self.results = {}
    
    def detect_service(self, port):
        """检测端口是否开放"""
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(self.timeout)
        try:
            result = sock.connect_ex((self.host, port))
            sock.close()
            return result == 0
        except:
            return False
    
    def scan_all(self):
        """扫描所有支持的服务"""
        print(f"弱口令检测工具")
        print(f"目标主机：{self.host}")
        print("=" * 50)
        
        # 第一阶段：检测开放端口
        print("\n[阶段1] 检测可用服务...")
        available_services = []
        
        for service_name, service_info in self.SUPPORTED_SERVICES.items():
            port = service_info["port"]
            if self.detect_service(port):
                available_services.append(service_name)
                print(f"  [开放] {service_name} ({service_info['description']}) - 端口 {port}")
            else:
                print(f"  [关闭] {service_name} - 端口 {port}")
        
        if not available_services:
            print("未发现开放的服务端口")
            return self.results
        
        # 第二阶段：执行弱口令检测
        print("\n[阶段2] 执行弱口令检测...")
        
        for service_name in available_services:
            service_info = self.SUPPORTED_SERVICES[service_name]
            print(f"\n--- 检测 {service_name} ({service_info['description']}) ---")
            
            try:
                cracker_class = service_info["class"]
                cracker = cracker_class(self.host, service_info["port"], self.timeout)
                
                start_time = time.time()
                found_creds = cracker.crack(max_workers=5)
                duration = time.time() - start_time
                
                self.results[service_name] = {
                    "port": service_info["port"],
                    "found": len(found_creds),
                    "credentials": found_creds,
                    "duration": round(duration, 2)
                }
            except ImportError as e:
                print(f"  缺少依赖：{e}，跳过检测")
                self.results[service_name] = {
                    "error": f"缺少依赖库：{e}"
                }
            except Exception as e:
                print(f"  检测异常：{e}")
                self.results[service_name] = {
                    "error": str(e)
                }
        
        # 输出报告
        self._print_report()
        return self.results
    
    def _print_report(self):
        """打印检测报告"""
        print("\n" + "=" * 50)
        print("弱口令检测报告")
        print("=" * 50)
        print(f"目标：{self.host}")
        print(f"检测时间：{time.strftime('%Y-%m-%d %H:%M:%S')}")
        print("-" * 50)
        
        total_found = 0
        for service, result in self.results.items():
            if "error" in result:
                print(f"\n{service.upper()}：检测失败 - {result['error']}")
            elif result["found"] > 0:
                total_found += result["found"]
                print(f"\n{service.upper()}：发现 {result['found']} 组弱口令！")
                for user, pwd in result["credentials"]:
                    print(f"  {user}:{pwd}")
                print(f"  耗时：{result['duration']}秒")
            else:
                print(f"\n{service.upper()}：未发现弱口令")
        
        if total_found > 0:
            print(f"\n总共发现 {total_found} 组弱口令，请立即修复！")
        else:
            print("\n未发现弱口令")
        
        print("=" * 50)
    
    def export_json(self, filename="weak_password_report.json"):
        """导出JSON报告"""
        with open(filename, "w", encoding="utf-8") as f:
            json.dump({
                "target": self.host,
                "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
                "results": self.results
            }, f, ensure_ascii=False, indent=2)
        print(f"报告已导出至 {filename}")


def main():
    """主函数"""
    import argparse
    
    parser = argparse.ArgumentParser(description="弱口令检测工具")
    parser.add_argument("host", help="目标主机地址")
    parser.add_argument("--timeout", type=int, default=5, help="超时时间")
    parser.add_argument("--json", help="导出JSON报告路径")
    parser.add_argument("--services", nargs="+", 
                       choices=["ssh", "ftp", "mysql", "redis"],
                       help="指定检测的服务")
    
    args = parser.parse_args()
    
    detector = WeakPasswordDetector(args.host, args.timeout)
    
    if args.services:
        # 仅检测指定服务
        detector.SUPPORTED_SERVICES = {
            s: detector.SUPPORTED_SERVICES[s] for s in args.services
        }
    
    detector.scan_all()
    
    if args.json:
        detector.export_json(args.json)


if __name__ == "__main__":
    main()
```
