# 多线程与多进程

## 5.1 多线程基础

多线程允许程序在同一时间执行多个任务。在安全工具中，常用于并发扫描以提高效率。

### threading 模块

```python
import threading
import time

def worker(name, delay):
    """线程工作函数"""
    for i in range(3):
        time.sleep(delay)
        print(f"[线程 {name}] 第{i+1}次执行")

# 创建线程
t1 = threading.Thread(target=worker, args=("A", 0.5))
t2 = threading.Thread(target=worker, args=("B", 0.8))

# 启动线程
t1.start()
t2.start()

# 等待线程完成
t1.join()
t2.join()

print("所有线程执行完毕")
```

## 5.2 ThreadPoolExecutor 线程池

使用线程池可以更高效地管理大量线程：

```python
from concurrent.futures import ThreadPoolExecutor
import time

def scan_port(port):
    """模拟端口扫描"""
    time.sleep(1)  # 模拟网络延迟
    return port, f"端口{port}状态：开放"

# 创建线程池，最大并发5个线程
ports = [80, 443, 8080, 22, 21, 3306, 6379, 8443]

with ThreadPoolExecutor(max_workers=5) as executor:
    # 提交所有任务
    futures = [executor.submit(scan_port, port) for port in ports]
    
    # 收集结果（按完成顺序）
    for future in futures:
        port, result = future.result()
        print(result)

print("批量扫描完成")
```

## 5.3 线程同步

多个线程同时访问共享资源时，需要使用锁机制避免数据竞争。

### Lock（互斥锁）

```python
import threading

results = []
results_lock = threading.Lock()
counter = 0
counter_lock = threading.Lock()

def safe_scan(port):
    global counter
    # 模拟扫描
    result = f"端口{port}已扫描"
    
    # 线程安全的操作
    with results_lock:
        results.append(result)
    
    with counter_lock:
        counter += 1

ports = range(1, 101)
threads = [threading.Thread(target=safe_scan, args=(p,)) for p in ports]

for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"扫描完成：{counter} 个端口")
print(f"结果数量：{len(results)}")
```

### 生产者-消费者模式

```python
import threading
import queue
import time

# 任务队列
task_queue = queue.Queue()
result_queue = queue.Queue()

def producer():
    """生产者：产生扫描任务"""
    for port in range(1, 1001):
        task_queue.put(port)
    print("所有任务已提交")

def worker():
    """消费者：执行扫描"""
    while not task_queue.empty():
        try:
            port = task_queue.get(timeout=1)
            time.sleep(0.001)  # 模拟扫描
            result_queue.put(f"端口{port}完成")
            task_queue.task_done()
        except queue.Empty:
            break

# 启动1个生产者、5个消费者
threads = []
threads.append(threading.Thread(target=producer))
for _ in range(5):
    threads.append(threading.Thread(target=worker))

for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"所有扫描完成，共处理 {result_queue.qsize()} 个结果")
```

## 5.4 多进程

对于CPU密集型任务，多进程比多线程更有效，可以充分利用多核CPU。

### multiprocessing 模块

```python
import multiprocessing
import time

def cpu_intensive(n):
    """CPU密集型计算"""
    total = 0
    for i in range(n):
        total += i ** 2
    return total

# 单进程
start = time.time()
results = [cpu_intensive(10**7) for _ in range(4)]
print(f"单进程耗时：{time.time() - start:.2f}s")

# 多进程
start = time.time()
with multiprocessing.Pool(4) as pool:
    results = pool.map(cpu_intensive, [10**7] * 4)
print(f"多进程耗时：{time.time() - start:.2f}s")
```

## 5.5 多线程 vs 多进程

| 特性 | 多线程 (threading) | 多进程 (multiprocessing) |
|------|-------------------|------------------------|
| 内存 | 共享内存（需锁） | 独立内存（无需锁） |
| 适用 | I/O密集型任务 | CPU密集型任务 |
| 创建开销 | 较小 | 较大 |
| 通信 | 共享变量需同步 | Queue/Pipe |
| GIL影响 | 受影响 | 不受影响 |

## 5.6 实际应用：并发端口扫描

```python
import socket
import threading
from concurrent.futures import ThreadPoolExecutor
import time

class PortScanner:
    """并发端口扫描器"""
    
    def __init__(self, host, timeout=2, max_workers=100):
        self.host = host
        self.timeout = timeout
        self.max_workers = max_workers
        self.open_ports = []
        self.lock = threading.Lock()
    
    def scan_port(self, port):
        """扫描单个端口"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            result = sock.connect_ex((self.host, port))
            sock.close()
            
            if result == 0:
                with self.lock:
                    self.open_ports.append(port)
                return port, "open"
            return port, "closed"
        except Exception:
            return port, "error"
    
    def scan_ports(self, ports):
        """扫描端口列表"""
        print(f"开始扫描 {self.host}，共 {len(ports)} 个端口")
        start_time = time.time()
        
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [executor.submit(self.scan_port, p) for p in ports]
            for future in futures:
                future.result()  # 等待所有任务完成
        
        duration = time.time() - start_time
        return self.open_ports, duration
    
    def scan_range(self, start_port, end_port):
        """扫描端口范围"""
        ports = list(range(start_port, end_port + 1))
        return self.scan_ports(ports)

# 使用示例
if __name__ == "__main__":
    scanner = PortScanner("scanme.nmap.org", timeout=2, max_workers=50)
    open_ports, duration = scanner.scan_range(1, 1024)
    
    print(f"\n扫描耗时：{duration:.2f}秒")
    print(f"开放端口（共{len(open_ports)}个）：")
    for port in sorted(open_ports):
        print(f"  {port}")
```

# 2. 编写一个进度跟踪器
class ProgressTracker:
    """跟踪多线程任务进度"""
    
    def __init__(self, total):
        self.total = total
        self.completed = 0
        self.lock = threading.Lock()
    
    def advance(self, n=1):
        with self.lock:
            self.completed += n
            pct = self.completed / self.total * 100
            print(f"\r进度：{self.completed}/{self.total} ({pct:.1f}%)", end="")
