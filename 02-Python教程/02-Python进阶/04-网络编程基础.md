# 网络编程基础

## 4.1 TCP/IP协议基础

TCP/IP是互联网通信的核心协议。理解其基本原理对开发网络工具至关重要。

- **IP地址**：标识网络中的设备
- **端口号**：标识设备上的具体服务（0-65535）
- **TCP**：面向连接的可靠传输协议（如HTTP、SSH）
- **UDP**：无连接的不可靠传输协议（如DNS、DHCP）

## 4.2 socket 编程详解

### TCP客户端完整示例

```python
import socket

def http_request(host, port=80, path="/"):
    """
    发送原始HTTP请求
    
    Args:
        host: 目标主机
        port: 目标端口
        path: 请求路径
    
    Returns:
        服务器响应字符串
    """
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(5)
    
    try:
        sock.connect((host, port))
        
        # 构造HTTP请求
        request = (
            f"GET {path} HTTP/1.1\r\n"
            f"Host: {host}\r\n"
            f"User-Agent: Mozilla/5.0\r\n"
            f"Connection: close\r\n"
            f"\r\n"
        )
        sock.send(request.encode())
        
        # 接收响应
        response = b""
        while True:
            data = sock.recv(4096)
            if not data:
                break
            response += data
        
        return response.decode("utf-8", errors="ignore")
    
    except socket.timeout:
        return f"请求 {host}:{port} 超时"
    except Exception as e:
        return f"异常：{e}"
    finally:
        sock.close()

response = http_request("example.com", 80, "/")
print(response[:500])  # 打印前500个字符
```

### TCP服务器完整示例

```python
import socket
import threading

def handle_client(client_sock, addr):
    """处理客户端连接"""
    print(f"新连接：{addr}")
    try:
        data = client_sock.recv(1024)
        response = f"收到 {len(data)} 字节数据\n".encode()
        client_sock.send(response)
    except Exception as e:
        print(f"处理异常：{e}")
    finally:
        client_sock.close()

def start_server(host="0.0.0.0", port=8888):
    """启动TCP服务器"""
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server.bind((host, port))
    server.listen(10)
    print(f"服务器启动，监听 {host}:{port}")
    
    try:
        while True:
            client, addr = server.accept()
            thread = threading.Thread(target=handle_client, args=(client, addr))
            thread.start()
    except KeyboardInterrupt:
        print("服务器关闭")
    finally:
        server.close()

# start_server()  # 取消注释可运行
```

## 4.3 UDP 通信

### UDP客户端

```python
import socket

def udp_client(message, host="127.0.0.1", port=9999):
    sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    try:
        sock.sendto(message.encode(), (host, port))
        data, addr = sock.recvfrom(1024)
        return data.decode()
    finally:
        sock.close()
```

### UDP服务器

```python
import socket

def udp_server(host="0.0.0.0", port=9999):
    sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    sock.bind((host, port))
    print(f"UDP服务器启动，监听 {host}:{port}")
    
    while True:
        data, addr = sock.recvfrom(1024)
        print(f"收到来自 {addr} 的数据：{data.decode()}")
        sock.sendto(b"ACK", addr)
```

## 4.4 DNS 解析

```python
import socket

def dns_lookup(hostname):
    """获取域名对应的IP地址"""
    try:
        ip = socket.gethostbyname(hostname)
        print(f"{hostname} -> {ip}")
        return ip
    except socket.gaierror:
        print(f"无法解析域名：{hostname}")
        return None

def dns_lookup_all(hostname):
    """获取所有IP地址"""
    try:
        results = socket.getaddrinfo(hostname, 80)
        ips = set()
        for result in results:
            ips.add(result[4][0])
        print(f"{hostname} 的IP地址：{ips}")
        return ips
    except socket.gaierror:
        print(f"无法解析域名：{hostname}")
        return None

dns_lookup("example.com")
dns_lookup_all("google.com")
```

## 4.5 端口状态检测

```python
import socket

def check_port(host, port, timeout=2):
    """
    检测端口状态
    
    Returns:
        "open": 端口开放
        "closed": 端口关闭
        "filtered": 端口被过滤
    """
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    
    try:
        result = sock.connect_ex((host, port))
        if result == 0:
            return "open"
        else:
            return "closed"
    except socket.timeout:
        return "filtered"
    except socket.gaierror:
        return "unknown_host"
    except Exception:
        return "error"
    finally:
        sock.close()

# 测试常用端口
host = "scanme.nmap.org"
for port in [22, 80, 443, 8080]:
    status = check_port(host, port)
    print(f"{host}:{port} - {status}")
```

## 4.7 网络字节序转换

```python
import socket

# 主机序与网络序转换
port = 80
network_port = socket.htons(port)  # 主机序转网络序
host_port = socket.ntohs(network_port)  # 网络序转主机序
print(f"端口 {port} 的网络序：{network_port}")

ip = "192.168.1.1"
packed_ip = socket.inet_aton(ip)  # IP转32位字节
unpacked_ip = socket.inet_ntoa(packed_ip)  # 字节转IP
print(f"IP {ip} 的打包形式：{packed_ip}")
```

## 4.8 实际应用：简易HTTP服务器

```python
import socket
import threading

def simple_http_server(host="0.0.0.0", port=8080):
    """极简HTTP服务器"""
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server.bind((host, port))
    server.listen(5)
    print(f"HTTP服务器启动：http://{host}:{port}")
    
    while True:
        client, addr = server.accept()
        data = client.recv(1024).decode()
        print(f"请求：{data.split(chr(13))[0] if data else '空请求'}")
        
        response = """HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\n
        <html><body><h1>Hello from Python!</h1></body></html>"""
        client.send(response.encode())
        client.close()

```
