# 端口扫描器

## 2.1 端口扫描原理

端口是计算机与外界通信的通道。端口扫描用于探测目标主机上哪些端口处于监听状态。

### 端口状态

| 状态 | 含义 |
|------|------|
| **open** | 端口开放，有应用程序在此监听 |
| **closed** | 端口关闭，无应用程序监听 |
| **filtered** | 端口被防火墙过滤，无法判断状态 |
| **unfiltered** | 端口可达但无法判断开放/关闭 |

### 常见端口号范围

- 0-1023：系统端口（Well Known Ports）
- 1024-49151：注册端口（Registered Ports）
- 49152-65535：动态/私有端口（Dynamic Ports）

## 2.2 端口扫描方式对比

Python中常用的端口扫描方式：

| 方式 | 速度 | 准确性 | 隐蔽性 | 权限需求 |
|------|------|--------|--------|---------|
| TCP Connect | 慢 | 高 | 低 | 无 |
| SYN Scan | 快 | 高 | 中 | 需原始socket |
| FIN Scan | 慢 | 中 | 高 | 需原始socket |
| UDP Scan | 慢 | 低 | 高 | 无 |

## 2.3 高级端口扫描器

```python
import socket
import threading
import time
import concurrent.futures
from dataclasses import dataclass
from typing import List, Optional
import json

@dataclass
class PortResult:
    """端口扫描结果数据结构"""
    port: int
    state: str  # open, closed, filtered
    service: str = ""
    banner: Optional[str] = None

class AdvancedPortScanner:
    """
    高级端口扫描器
    
    支持：
    - TCP Connect扫描
    - 多线程并发
    - 超时控制
    - 服务识别
    - 结果导出
    """
    
    # 常见端口与服务映射
    COMMON_PORTS = {
        21: "FTP", 22: "SSH", 23: "Telnet", 25: "SMTP",
        53: "DNS", 80: "HTTP", 110: "POP3", 135: "RPC",
        139: "NetBIOS", 143: "IMAP", 443: "HTTPS", 445: "SMB",
        993: "IMAPS", 995: "POP3S", 1433: "MSSQL", 1521: "Oracle",
        2049: "NFS", 3306: "MySQL", 3389: "RDP", 5432: "PostgreSQL",
        5900: "VNC", 5985: "WinRM-HTTP", 5986: "WinRM-HTTPS",
        6379: "Redis", 8080: "HTTP-Proxy", 8443: "HTTPS-Alt",
        27017: "MongoDB", 33890: "RDP-Alt"
    }
    
    def __init__(self, host: str, timeout: float = 2.0, 
                 max_workers: int = 100, use_banner_grab: bool = True):
        self.host = host
        self.timeout = timeout
        self.max_workers = max_workers
        self.use_banner_grab = use_banner_grab
        self.ip = self._resolve_host()
        if not self.ip:
            raise ValueError(f"无法解析主机：{host}")
    
    def _resolve_host(self) -> Optional[str]:
        """解析主机名"""
        try:
            return socket.gethostbyname(self.host)
        except socket.gaierror:
            return None
    
    def _tcp_connect(self, port: int) -> str:
        """
        TCP Connect扫描
        
        发送SYN包，收到SYN-ACK表示开放，收到RST表示关闭
        """
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(self.timeout)
        
        try:
            result = sock.connect_ex((self.ip, port))
            if result == 0:
                self._send_rst(sock)  # 优雅关闭
                return "open"
            return "closed"
        except socket.timeout:
            return "filtered"
        except Exception:
            return "error"
        finally:
            try:
                sock.close()
            except:
                pass
    
    def _send_rst(self, sock: socket.socket):
        """发送RST包关闭连接"""
        try:
            sock.setblocking(False)
            sock.send(b"")
        except:
            pass
    
    def _grab_banner(self, port: int) -> Optional[str]:
        """抓取服务横幅"""
        if not self.use_banner_grab:
            return None
        
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(self.timeout)
        
        try:
            sock.connect((self.ip, port))
            
            # 部分服务需要发送请求
            if port in [80, 8080, 8000]:
                sock.send(b"GET / HTTP/1.0\r\nHost: example.com\r\n\r\n")
            elif port in [21]:
                pass  # FTP主动发送横幅
            elif port in [25]:
                sock.send(b"EHLO scan\r\n")
            elif port in [110]:
                sock.send(b"USER test\r\n")
            
            time.sleep(0.3)  # 等待响应
            banner = sock.recv(256).decode("utf-8", errors="ignore").strip()
            return banner[:200]  # 限制长度
        
        except:
            return None
        finally:
            sock.close()
    
    def _guess_service(self, port: int) -> str:
        """根据端口号猜测服务名称"""
        return self.COMMON_PORTS.get(port, "unknown")
    
    def scan_single_port(self, port: int) -> PortResult:
        """扫描单个端口"""
        state = self._tcp_connect(port)
        service = self._guess_service(port)
        banner = None
        
        if state == "open" and self.use_banner_grab:
            banner = self._grab_banner(port)
        
        return PortResult(
            port=port,
            state=state,
            service=service,
            banner=banner
        )
    
    def scan_ports(self, ports: List[int]) -> List[PortResult]:
        """并发扫描端口列表"""
        results = []
        scanned = 0
        total = len(ports)
        start_time = time.time()
        
        print(f"扫描目标：{self.host} ({self.ip})")
        print(f"端口数量：{total}")
        print(f"最大并发：{self.max_workers}")
        print("-" * 40)
        
        with concurrent.futures.ThreadPoolExecutor(
            max_workers=self.max_workers
        ) as executor:
            future_to_port = {
                executor.submit(self.scan_single_port, port): port 
                for port in ports
            }
            
            for future in concurrent.futures.as_completed(future_to_port):
                scanned += 1
                result = future.result()
                
                # 实时进度显示
                if scanned % 50 == 0 or result.state == "open":
                    pct = scanned / total * 100
                    print(f"\r进度：{scanned}/{total} ({pct:.1f}%)", end="")
                
                results.append(result)
        
        duration = time.time() - start_time
        print(f"\n扫描完成，耗时：{duration:.2f}秒")
        
        return sorted(results, key=lambda x: x.port)
    
    def scan_range(self, start: int, end: int) -> List[PortResult]:
        """扫描端口范围"""
        ports = list(range(start, end + 1))
        return self.scan_ports(ports)
    
    def scan_common(self) -> List[PortResult]:
        """仅扫描常见端口"""
        return self.scan_ports(list(self.COMMON_PORTS.keys()))
    
    def print_report(self, results: List[PortResult]):
        """打印扫描报告"""
        open_ports = [r for r in results if r.state == "open"]
        
        print("\n" + "=" * 50)
        print("端口扫描报告")
        print("=" * 50)
        print(f"目标主机：{self.host} ({self.ip})")
        print(f"扫描端口：{len(results)} 个")
        print(f"开放端口：{len(open_ports)} 个")
        print("-" * 50)
        
        if open_ports:
            print(f"{'端口':<8} {'状态':<10} {'服务':<15} {'横幅'}")
            print("-" * 50)
            for r in open_ports:
                banner = r.banner[:40] if r.banner else ""
                print(f"{r.port:<8} {r.state:<10} {r.service:<15} {banner}")
        
        print("=" * 50)
    
    def export_json(self, results: List[PortResult], 
                    filename: str = "port_scan_result.json"):
        """导出JSON格式报告"""
        data = {
            "target": self.host,
            "ip": self.ip,
            "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
            "total_ports": len(results),
            "open_ports": len([r for r in results if r.state == "open"]),
            "results": [
                {
                    "port": r.port,
                    "state": r.state,
                    "service": r.service,
                    "banner": r.banner
                }
                for r in results
            ]
        }
        
        with open(filename, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        print(f"报告已导出：{filename}")


def main():
    """主函数"""
    import argparse
    
    parser = argparse.ArgumentParser(
        description="Advanced Port Scanner",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
示例：
  python port_scanner.py example.com
  python port_scanner.py 192.168.1.1 -p 1-1024
  python port_scanner.py scanme.nmap.org --timeout 3 --workers 200
        """
    )
    
    parser.add_argument("host", help="目标主机/IP")
    parser.add_argument("-p", "--ports", default="common",
                       help="端口范围，如: 1-1024, 80,443,8080 或 common")
    parser.add_argument("--timeout", type=float, default=2.0,
                       help="超时时间（秒）")
    parser.add_argument("--workers", type=int, default=100,
                       help="最大并发线程数")
    parser.add_argument("--no-banner", action="store_true",
                       help="不抓取服务横幅")
    parser.add_argument("--json", default="",
                       help="输出JSON文件路径")
    
    args = parser.parse_args()
    
    try:
        scanner = AdvancedPortScanner(
            host=args.host,
            timeout=args.timeout,
            max_workers=args.workers,
            use_banner_grab=not args.no_banner
        )
        
        # 解析端口参数
        if args.ports == "common":
            results = scanner.scan_common()
        elif "-" in args.ports:
            start, end = args.ports.split("-")
            results = scanner.scan_range(int(start), int(end))
        else:
            ports = [int(p.strip()) for p in args.ports.split(",")]
            results = scanner.scan_ports(ports)
        
        scanner.print_report(results)
        
        if args.json:
            scanner.export_json(results, args.json)
    
    except ValueError as e:
        print(f"错误：{e}")
        return 1
    except KeyboardInterrupt:
        print("\n扫描被用户中断")
        return 1
    
    return 0


if __name__ == "__main__":
    exit(main())
```

## 2.4 UDP端口扫描

```python
import socket

def udp_port_scan(host: str, ports: list, timeout: float = 3.0) -> list:
    """
    UDP端口扫描
    
    UDP扫描原理：
    - 发送UDP数据包到目标端口
    - 收到ICMP Port Unreachable → 端口关闭
    - 无响应或收到其他响应 → 端口可能开放
    - 收到响应数据 → 端口开放
    """
    open_ports = []
    
    for port in ports:
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            sock.settimeout(timeout)
            
            # 发送UDP数据
            sock.sendto(b"\x00", (host, port))
            
            try:
                data, addr = sock.recvfrom(1024)
                open_ports.append((port, "open"))
                print(f"[开放] {port}/udp - 收到响应")
            except socket.timeout:
                # 无响应可能意味着端口开放或过滤
                print(f"[可能开放] {port}/udp - 无响应")
            
            sock.close()
        
        except socket.error as e:
            if "Connection refused" in str(e):
                print(f"[关闭] {port}/udp")
            else:
                print(f"[错误] {port}/udp - {e}")
    
    return open_ports
```

## 2.5 端口扫描性能优化

```python
import time
import socket
import concurrent.futures

def benchmark_scan_methods(host: str):
    """对比不同扫描方式的性能"""
    
    def scan_with_limit(ports, max_workers):
        """带并发限制的扫描"""
        open_ports = []
        
        def check(port):
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(2)
            try:
                result = sock.connect_ex((host, port))
                return port, result == 0
            finally:
                sock.close()
        
        with concurrent.futures.ThreadPoolExecutor(max_workers) as ex:
            futures = {ex.submit(check, p): p for p in ports}
            for future in concurrent.futures.as_completed(futures):
                port, is_open = future.result()
                if is_open:
                    open_ports.append(port)
        
        return sorted(open_ports)
    
    test_ports = list(range(1, 1001))
    
    for workers in [10, 50, 100, 200, 500]:
        start = time.time()
        results = scan_with_limit(test_ports, workers)
        duration = time.time() - start
        print(f"并发数 {workers:3d} → 耗时 {duration:.2f}s, 发现 {len(results)} 个开放端口")
```
