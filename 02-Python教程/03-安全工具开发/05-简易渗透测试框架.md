## 5.2 模块管理器

```python
import importlib
import inspect
import os
import json
import time
from typing import Dict, List, Any, Optional

class ModuleManager:
    """
    模块管理器
    
    负责加载、注册、调用各个功能模块
    """
    
    def __init__(self):
        self.modules = {}
        self.results = {}
    
    def register_module(self, name: str, module_class, description: str = ""):
        """注册模块"""
        self.modules[name] = {
            "class": module_class,
            "description": description,
            "instance": None
        }
    
    def get_module(self, name: str):
        """获取模块实例"""
        if name not in self.modules:
            raise KeyError(f"模块 '{name}' 未注册")
        if self.modules[name]["instance"] is None:
            raise RuntimeError(f"模块 '{name}' 未初始化")
        return self.modules[name]["instance"]
    
    def list_modules(self) -> List[Dict]:
        """列出所有已注册的模块"""
        modules_info = []
        for name, info in self.modules.items():
            modules_info.append({
                "name": name,
                "description": info["description"],
                "loaded": info["instance"] is not None
            })
        return modules_info
    
    def init_module(self, name: str, *args, **kwargs):
        """初始化指定模块"""
        if name not in self.modules:
            raise KeyError(f"模块 '{name}' 未注册")
        module_class = self.modules[name]["class"]
        self.modules[name]["instance"] = module_class(*args, **kwargs)
    
    def init_all(self, default_args: Dict = None):
        """初始化所有模块"""
        if default_args is None:
            default_args = {}
        for name in self.modules:
            if self.modules[name]["instance"] is None:
                self.init_module(name, **default_args)
```

## 5.3 任务调度器

```python
from enum import Enum
from dataclasses import dataclass, field
from typing import Callable

class TaskStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"

@dataclass
class Task:
    """任务单元"""
    name: str
    func: Callable
    args: tuple = ()
    kwargs: dict = field(default_factory=dict)
    status: TaskStatus = TaskStatus.PENDING
    result: Any = None
    error: Optional[str] = None
    duration: float = 0.0

class TaskScheduler:
    """
    任务调度器
    
    管理任务的执行顺序和依赖关系
    """
    
    def __init__(self):
        self.tasks: List[Task] = []
        self.task_order = []  # 按执行顺序排列的任务名
    
    def add_task(self, task: Task, order: int = 0):
        """添加任务"""
        self.tasks.append(task)
        self.task_order.append((order, task.name))
        self.task_order.sort(key=lambda x: x[0])  # 按优先级排序
    
    def run_all(self) -> Dict[str, Any]:
        """按顺序执行所有任务"""
        results = {}
        total = len(self.task_order)
        
        print("任务调度器开始执行")
        print("=" * 40)
        
        for idx, (order, task_name) in enumerate(self.task_order, 1):
            task = next(t for t in self.tasks if t.name == task_name)
            
            print(f"[{idx}/{total}] 执行任务：{task.name}")
            task.status = TaskStatus.RUNNING
            
            start = time.time()
            try:
                task.result = task.func(*task.args, **task.kwargs)
                task.status = TaskStatus.COMPLETED
                task.duration = time.time() - start
                print(f"  -> 完成，耗时：{task.duration:.2f}s")
            except Exception as e:
                task.status = TaskStatus.FAILED
                task.error = str(e)
                task.duration = time.time() - start
                print(f"  -> 失败：{e}")
            
            results[task_name] = {
                "status": task.status.value,
                "result": task.result,
                "error": task.error,
                "duration": task.duration
            }
        
        print("=" * 40)
        return results
```

## 5.5 配置管理器

```python
import json
import os
from typing import Dict, Any

class ConfigManager:
    """配置文件管理器"""
    
    DEFAULT_CONFIG = {
        "scanner": {
            "timeout": 3,
            "max_workers": 100,
            "common_ports_only": True
        },
        "cracker": {
            "timeout": 5,
            "max_workers": 10,
            "wordlist": "wordlists/weak_passwords.txt"
        },
        "web_collector": {
            "timeout": 5,
            "max_urls": 20,
            "dir_wordlist": "wordlists/common_dirs.txt"
        },
        "report": {
            "format": "html",
            "output_dir": "output/"
        }
    }
    
    def __init__(self, config_path: str = "config/config.json"):
        self.config_path = config_path
        self.config = self.load()
    
    def load(self) -> Dict[str, Any]:
        """加载配置文件"""
        if os.path.exists(self.config_path):
            with open(self.config_path, "r", encoding="utf-8") as f:
                return json.load(f)
        return self.DEFAULT_CONFIG.copy()
    
    def save(self):
        """保存配置"""
        os.makedirs(os.path.dirname(self.config_path), exist_ok=True)
        with open(self.config_path, "w", encoding="utf-8") as f:
            json.dump(self.config, f, ensure_ascii=False, indent=2)
    
    def get(self, *keys):
        """获取配置值，支持嵌套键"""
        value = self.config
        for key in keys:
            if isinstance(value, dict):
                value = value.get(key)
            else:
                return None
        return value
    
    def set(self, *keys, value):
        """设置配置值"""
        current = self.config
        for key in keys[:-1]:
            if key not in current:
                current[key] = {}
            current = current[key]
        current[keys[-1]] = value
```

## 5.6 综合框架入口

```python
import sys
import os
import json

class PentestFramework:
    """
    渗透测试综合框架
    
    整合扫描、弱口令检测、信息收集等功能
    """
    
    def __init__(self, target: str, config_path: str = "config/config.json"):
        self.target = target
        self.config = ConfigManager(config_path)
        self.module_manager = ModuleManager()
        self.task_scheduler = TaskScheduler()
        self.results = {}
    
    def setup(self):
        """初始化框架配置"""
        print("=" * 50)
        print("PenTest Framework")
        print("=" * 50)
        print(f"目标：{self.target}")
        
        # 创建输出目录
        output_dir = self.config.get("report", "output_dir") or "output"
        os.makedirs(output_dir, exist_ok=True)
    
    def run_port_scan(self):
        """端口扫描任务"""
        print("\n>>> 端口扫描")
        scanner = AdvancedPortScanner(
            host=self.target,
            timeout=self.config.get("scanner", "timeout") or 3,
            max_workers=self.config.get("scanner", "max_workers") or 100
        )
        
        if self.config.get("scanner", "common_ports_only"):
            results = scanner.scan_common()
        else:
            results = scanner.scan_range(1, 1024)
        
        scanner.print_report(results)
        open_ports = [
            {"port": r.port, "service": r.service, "banner": r.banner}
            for r in results if r.state == "open"
        ]
        
        return {"results": results, "open_ports": open_ports}
    
    def run_web_info_collect(self, port: int = 80):
        """Web信息收集任务"""
        print(f"\n>>> Web信息收集（端口：{port}）")
        base_url = f"http://{self.target}:{port}" if port != 80 else f"http://{self.target}"
        
        collector = WebInfoCollector(base_url)
        info = collector.collect()
        return info
    
    def run_weak_password_check(self, services: list = None):
        """弱口令检测任务"""
        print("\n>>> 弱口令检测")
        detector = WeakPasswordDetector(
            self.target,
            timeout=self.config.get("cracker", "timeout") or 5
        )
        
        if services:
            detector.SUPPORTED_SERVICES = {
                s: detector.SUPPORTED_SERVICES[s] for s in services
            }
        
        return detector.scan_all()
    
    def run_full_scan(self):
        """执行完整扫描流程"""
        self.setup()
        
        # 1. 端口扫描
        print("\n[阶段 1/3] 端口扫描")
        scan_result = self.run_port_scan()
        self.results["port_scan"] = scan_result
        
        # 2. Web信息收集（如果发现HTTP服务）
        web_ports = [
            p["port"] for p in scan_result["open_ports"]
            if p["port"] in [80, 443, 8080, 8443]
        ]
        
        if web_ports:
            print("\n[阶段 2/3] Web信息收集")
            web_info = self.run_web_info_collect(web_ports[0])
            self.results["web_info"] = web_info
        
        # 3. 弱口令检测
        print("\n[阶段 3/3] 弱口令检测")
        available_services = []
        for r in scan_result["open_ports"]:
            service_map = {22: "ssh", 21: "ftp", 3306: "mysql", 6379: "redis"}
            if r["port"] in service_map:
                available_services.append(service_map[r["port"]])
        
        if available_services:
            pwd_result = self.run_weak_password_check(available_services)
            self.results["weak_password"] = pwd_result
        
        # 4. 生成报告
        self._generate_report()
        
        return self.results
    
    def _generate_report(self):
        """生成测试报告"""
        open_ports = self.results.get("port_scan", {}).get("open_ports", [])
        weak_passwords = []
        
        wp_result = self.results.get("weak_password", {})
        for service, info in wp_result.items():
            if isinstance(info, dict) and info.get("credentials"):
                for cred in info["credentials"]:
                    weak_passwords.append([service.upper(), cred[0], cred[1]])
        
        directories = []
        web_info = self.results.get("web_info", {})
        if isinstance(web_info, dict):
            directories = web_info.get("directories", [])
        
        summary = {
            "target": self.target,
            "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
            "summary": {
                "open_ports": len(open_ports),
                "weak_passwords": len(weak_passwords),
                "directories": len(directories)
            },
            "details": {
                "open_ports": open_ports,
                "weak_passwords": weak_passwords,
                "directories": directories
            }
        }
        
        # 保存JSON报告
        output_dir = self.config.get("report", "output_dir") or "output"
        json_path = os.path.join(output_dir, f"report_{int(time.time())}.json")
        with open(json_path, "w", encoding="utf-8") as f:
            json.dump(summary, f, ensure_ascii=False, indent=2)
        
        # 保存HTML报告
        html = ReportGenerator.generate_html(self.results, self.target)
        html_path = os.path.join(output_dir, f"report_{int(time.time())}.html")
        ReportGenerator.save_report(html, html_path)
        
        print(f"\nJSON报告：{json_path}")
        print(f"HTML报告：{html_path}")


def main():
    """框架入口"""
    import argparse
    
    parser = argparse.ArgumentParser(
        description="PenTest Framework - 简易渗透测试框架"
    )
    parser.add_argument("target", help="目标主机")
    parser.add_argument("--mode", choices=["full", "port", "web", "password"],
                       default="full", help="扫描模式")
    parser.add_argument("--port", type=int, default=80,
                       help="Web服务端口")
    parser.add_argument("--services", nargs="+",
                       choices=["ssh", "ftp", "mysql", "redis"],
                       help="弱口令检测的服务类型")
    parser.add_argument("--config", default="config/config.json",
                       help="配置文件路径")
    
    args = parser.parse_args()
    
    framework = PentestFramework(args.target, args.config)
    
    if args.mode == "full":
        framework.run_full_scan()
    elif args.mode == "port":
        framework.setup()
        framework.run_port_scan()
    elif args.mode == "web":
        framework.setup()
        framework.run_web_info_collect(args.port)
    elif args.mode == "password":
        framework.setup()
        framework.run_weak_password_check(args.services)


if __name__ == "__main__":
    main()
```
